# <-- commentline
# /path/to/BIN;CMD Options(arg1,arg2,arg...);Path;Name;Icon
# AVOID using ";", this is the delimiter! use as example "&&" in bash commands
# For using own icons, copy icons to programs "icons" Folder and use iconname + extension, like "terminal.svg"
# If you want use Systemicons, set "usesystemicons = true" in config.ini AND use only icon name, like "folder"
# SSH
/usr/bin/terminator;--new-tab,-e;ssh <USERNAME@HOSTORIP>;<NAME>;terminal.svg

/usr/bin/codium;--folder-uri;vscode-remote://ssh-remote+<USERNAME>@<HOSTORIP><REMOTEFOLDER>;<NAME>;folder

/usr/bin/caja;-t;sftp://<pathtoyourfolder>;<NAME>;folder

/usr/bin/caja;-t;ftps://user:pass@<ftpurl>;<NAME>;folder
